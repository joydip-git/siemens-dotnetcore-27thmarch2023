        static bool TryToParse(string data, out DateTime dt)
        {
            bool status = false;
            try
            {
                dt = DateTime.Parse(data);
                status = true;
            }
            catch (Exception ex)
            {
                dt = DateTime.MinValue;
            }
            return status;
        }
